description: >
  <p>フレームバッファの自動サイズ調整モードを切り替えるか、現在のモードを返します。</p>

  <p>デフォルトでは、フレームバッファは自動的にサイズを調整して、それを作成したキャンバスに合わせます。

  <code>myBuffer.autoSized(false)</code>を呼び出すとこの動作を無効にし、

  <code>myBuffer.autoSized(true)</code>を呼び出すと再び有効にします。</p>

  <p><code>myBuffer.autoSized()</code>を引数なしで呼び出すと、フレームバッファが

  自動的にサイズ調整される場合は<code>true</code>を、そうでない場合は<code>false</code>を返します。</p>

    <code>
    // ダブルクリックで自動サイズ調整モードを切り替えます。

    let myBuffer;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // p5.Framebufferオブジェクトを作成します。
      myBuffer = createFramebuffer();

      describe('グレーの背景に多色の球体。ユーザーがマウスを動かすと画像のサイズが変わります。');
    }

    function draw() {
      background(50);

      // p5.Framebufferオブジェクトに描画します。
      myBuffer.begin();
      background(200);
      normalMaterial();
      sphere(width / 4);
      myBuffer.end();

      // p5.Framebufferオブジェクトを表示します。
      image(myBuffer, -width / 2, -height / 2);
    }

    // ユーザーがマウスを動かすとキャンバスのサイズを変更します。
    function mouseMoved() {
      let w = constrain(mouseX, 0, 100);
      let h = constrain(mouseY, 0, 100);
      resizeCanvas(w, h);
    }

    // ユーザーがダブルクリックするとautoSizingを切り替えます。
    // 注意：これを修正（？）するための問題を開きました。
    function doubleClicked() {
      let isAuto = myBuffer.autoSized();
      myBuffer.autoSized(!isAuto);
    }
