description: >
  <p>画像の着色を無効にします。</p>

  <p><code>tint()</code>は、後続の全ての<code>image()</code>呼び出しに影響します。
  <code>noTint()</code>を呼び出すと、画像は元の色で表示されます。</p>
line: 1
isConstructor: false
itemtype: method
example:
  - |-

    <div>
    <code>
    let img;

    // 画像を読み込みます。
    function preload() {
      img = loadImage('/assets/laDefense.jpg');
    }

    function setup() {
      createCanvas(100, 100);

      image(img, 0, 0);
      tint(0, 153, 204);  // 青色の色合い
      image(img, 50, 0);

      describe('2つの画像。左側は元の画像で、右側は青色の色合いがかかっています。');
    }
    </code>
    </div>

    <div>
    <code>
    let img;

    // 画像を読み込みます。
    function preload() {
      img = loadImage('/assets/laDefense.jpg');
    }

    function setup() {
      createCanvas(100, 100);
    }

    function draw() {
      image(img, 0, 0);
      if (mouseIsPressed) {
        tint(0, 153, 204);  // 青色の色合い
      } else {
        noTint();
      }
      image(img, 50, 0);

      describe('2つの画像。左側は元の画像で、右側はマウスが押されたときに青色の色合いがかかります。');
    }
    </code>
    </div>
