    function setup() {
      background(200);

      // コンテナとしてdiv要素を作成します。
      let div = createDiv();

      // divを中央に配置します。
      div.position(25, 35);

      // span要素を作成します。
      let s1 = createSpan('p5');

      // 2つ目のspan要素を作成します。
      let s2 = createSpan('*');

      // 2つ目のspanのフォントカラーを設定します。
      s2.style('color', 'deeppink');

      // 3つ目のspan要素を作成します。
      let s3 = createSpan('js');

      // すべてのspanをコンテナのdivに追加します。
      s1.parent(div);
      s2.parent(div);
      s3.parent(div);

      describe('中央に黒で"p5*js"と書かれたグレーの正方形。アスタリスクはピンク色です。');
    }
