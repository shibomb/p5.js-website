description: >
  <p>スケッチからすべてのライトを削除します。</p>

  <p><code>noLights()</code>を呼び出すと、

  <a href="/reference/p5/lights/">lights()</a>、

  <a href="/reference/p5/ambientLight/">ambientLight()</a>、

  <a href="/reference/p5/directionalLight/">directionalLight()</a>、

  <a href="/reference/p5/pointLight/">pointLight()</a>、または

  <a href="/reference/p5/spotLight/">spotLight()</a>で作成されたすべてのライトが削除されます。

  これらの関数は<code>noLights()</code>の後に呼び出して、新しい照明スキームを作成することができます。</p>

    // クリックしてドラッグすると、異なる角度からシーンを見ることができます。

    function setup() {
      createCanvas(100, 100, WEBGL);

      describe('灰色の背景に描かれた2つの球体。上の球体は白で、下の球体は赤です。');
    }

    function draw() {
      background(50);

      // マウスで軌道制御を有効にします。
      orbitControl();

      // ライトをオンにします。
      lights();

      // 球体のスタイルを設定します。
      noStroke();

      // 上の球体を描画します。
      push();
      translate(0, -25, 0);
      sphere(20);
      pop();

      // ライトをオフにします。
      noLights();

      // 画面内を指す赤い指向性ライトを追加します。
      directionalLight(255, 0, 0, 0, 0, -1);

      // 下の球体を描画します。
      push();
      translate(0, 25, 0);
      sphere(20);
      pop();
    }
