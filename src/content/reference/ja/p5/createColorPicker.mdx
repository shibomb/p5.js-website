  <p>カラーピッカー要素を作成します。</p>

  <p>パラメータ<code>value</code>はオプションです。色文字列または

  デフォルトの

  色を設定します。</p>

  <p>カラーピッカーは<a href="/reference/p5/p5.Element">p5.Element</a>
  クラスを拡張し、

  色を管理するための便利なメソッドをいくつか追加しています：</p>

  <ul>

  <li><code>myPicker.value()</code>は現在の色を<code>'#rrggbb'</code>形式の16進数文字列として返します。</li>

  <li><code>myPicker.color()</code>は現在の色を<a
  href="/reference/p5/p5.Color">p5.Color</a>オブジェクトとして返します。</li>

  </ul>

    let myPicker;

    function setup() {
      createCanvas(100, 100);

      // カラーピッカーを作成し、位置を設定します。
      myPicker = createColorPicker('deeppink');
      myPicker.position(0, 100);

      describe('"#rrggbb"形式の数字がピンク色のキャンバスに表示されています。ユーザーが新しい色を選ぶと、背景色と数字が変わります。');
    }

    function draw() {
      // カラーピッカーを使用して背景を塗ります。
      let c = myPicker.value();
      background(c);

      // 現在の色を16進数文字列として表示します。
      text(c, 25, 55);
    }
