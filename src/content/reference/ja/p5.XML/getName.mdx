  <p>要素の名前を<code>String</code>として返します。</p>

  <p>XML要素の名前はそのタグによって与えられます。例えば、

  <code><language>JavaScript</language></code>という要素の名前は
  <code>language</code>です。</p>
line: 128
isConstructor: false
itemtype: method
example:
  - |-

    <div>
    <code>
    let myXML;

    // XMLを読み込み、p5.XMLオブジェクトを作成します。
    function preload() {
      myXML = loadXML('/assets/animals.xml');
    }

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 全てのmammal要素を含む配列を取得します。
      let mammals = myXML.getChildren('mammal');

      // 最初のmammal要素を取得します。
      let firstMammal = mammals[0];

      // mammal要素の名前を取得します。
      let name = firstMammal.getName();

      // テキストのスタイルを設定します。
      textAlign(CENTER, CENTER);
      textFont('Courier New');
      textSize(14);

      // 要素の名前を表示します。
      text(name, 50, 50);

      describe('灰色の背景に黒で書かれた "mammal" という単語。');
    }
    </code>
    </div>
