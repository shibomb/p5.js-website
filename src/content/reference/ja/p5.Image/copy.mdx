description: >
  <p>ソース画像からこの画像にピクセルをコピーします。</p>

  <p>最初のパラメータ <code>srcImage</code> は、コピーする

  <a href="/reference/p5/p5.Image">p5.Image</a> オブジェクトのオプションです。ソース

  画像が渡されない場合、<code>img.copy()</code> はこの画像の一部を別の

  領域にコピーすることができます。</p>

  <p>次の4つのパラメータ、<code>sx</code>、<code>sy</code>、<code>sw</code>、<code>sh</code> は

  ソース画像からコピーする領域を決定します。<code>(sx, sy)</code> は領域の

  左上隅です。<code>sw</code> と <code>sh</code> は領域の幅と高さです。</p>

  <p>次の4つのパラメータ、<code>dx</code>、<code>dy</code>、<code>dw</code>、<code>dh</code> は

  この画像にコピーする領域を決定します。<code>(dx, dy)</code> は領域の

  左上隅です。<code>dw</code> と <code>dh</code> は領域の幅と高さです。</p>

  <p><code>img.copy()</code> を呼び出すと、ソース領域がコピー先の領域と

  同じサイズでない場合、ピクセルがスケーリングされます。</p>

    <code>
    let img;

    // 画像を読み込みます。
    function preload() {
      img = loadImage('/assets/rockies.jpg');
    }

    function setup() {
      createCanvas(100, 100);

      // 画像の一部の領域を別の領域にコピーします。
      img.copy(7, 22, 10, 10, 35, 25, 50, 50);

      // 画像を表示します。
      image(img, 0, 0);

      // コピーした領域を輪郭線で囲みます。
      stroke(255);
      noFill();
      square(7, 22, 10);

      describe('山の風景の画像。白い輪郭線で囲まれた正方形の領域がある。より大きな正方形には、輪郭線で囲まれた領域のピクセル化されたビューが含まれている。');
    }
    </code>

    let mountains;
    let bricks;

    // 画像を読み込みます。
    function preload() {
      mountains = loadImage('/assets/rockies.jpg');
      bricks = loadImage('/assets/bricks.jpg');
    }

    function setup() {
      createCanvas(100, 100);

      // レンガの画像の中心を計算します。
      let x = bricks.width / 2;
      let y = bricks.height / 2;

      // レンガを山の画像にコピーします。
      mountains.copy(bricks, 0, 0, x, y, 0, 0, x, y);

      // 山の画像を表示します。
      image(mountains, 0, 0);

      describe('山の風景の画像の左上に描かれたレンガの壁の画像。');
    }
